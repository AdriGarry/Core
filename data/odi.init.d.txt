#!/bin/bash

case "$1" in

    start)

        sudo sh /home/pi/odi/src/shell/led.sh allLedsOn

        sudo sh /home/pi/odi/src/shell/sounds.sh startup

        sudo sh /home/pi/odi/src/shell/led.sh allLedsOff

        sudo python /home/pi/odi/src/python/buttons.py 2>&1 | sudo tee -a /home/pi/odi/log/odi.log &

        sudo sh /home/pi/odi/odi.sh &

    ;;

    stop)

        sudo killall omxplayer.bin
        sudo killall node
        sudo sh /home/pi/odi/src/shell/sounds.sh reboot
        sudo sh /home/pi/odi/src/shell/led.sh allLedsOff
    ;;
esac

exit 0
